My reordering strategy focuses on further optimizing action order by placing `pour-shaker-to-shot` first (directly achieves the goal predicate `contains ?shot ?cocktail`), followed by `shake` (creates the cocktail in the shaker), then the pour-to-shaker actions and `fill-shot`/`refill-shot` that build preconditions, with `grasp`/`leave` and cleanup actions last; I also reorder predicates to place `contains` and `ontable` (goal-relevant) first, and within each action I reorder preconditions to place goal-proximal predicates before static/auxiliary ones, and effects to place goal-achieving add-effects before deletes. This tighter alignment of internal IDs with goal-relevance should produce more discriminating #r values in H_Add_Rp_Fwd, yielding better bucket assignments and improved novelty pruning.

(define (domain barman)
  (:requirements :strips :typing)
  (:types hand level beverage dispenser container - object
  	  ingredient cocktail - beverage
          shot shaker - container)
  (:predicates  (contains ?c - container ?b - beverage)
                (ontable ?c - container)
                (empty ?c - container)
                (clean ?c - container)
                (holding ?h - hand ?c - container)
                (handempty ?h - hand)
                (shaked ?s - shaker)
                (unshaked ?s - shaker)
                (used ?c - container ?b - beverage)
                (shaker-level ?s - shaker ?l - level)
                (shaker-empty-level ?s - shaker ?l - level)
                (next ?l1 ?l2 - level)
                (cocktail-part1 ?c - cocktail ?i - ingredient)
                (cocktail-part2 ?c - cocktail ?i - ingredient)
                (dispenses ?d - dispenser ?i - ingredient))

  (:action pour-shaker-to-shot
           :parameters (?b - beverage ?d - shot ?h - hand ?s - shaker ?l ?l1 - level)
           :precondition (and (contains ?s ?b)
                              (empty ?d)
                              (clean ?d)
                              (holding ?h ?s)
			      (shaked ?s)
                              (shaker-level ?s ?l)
                              (next ?l1 ?l))
           :effect (and (contains ?d ?b)
                        (shaker-level ?s ?l1)
                        (not (empty ?d))
                        (not (clean ?d))
			(not (shaker-level ?s ?l))))

  (:action shake
  	   :parameters (?b - cocktail ?d1 ?d2 - ingredient ?s - shaker ?h1 ?h2 - hand)
           :precondition (and (contains ?s ?d1)
                              (contains ?s ?d2)
                              (cocktail-part1 ?b ?d1)
			      (cocktail-part2 ?b ?d2)
                              (holding ?h1 ?s)
                              (handempty ?h2)
			      (unshaked ?s))
           :effect (and (contains ?s ?b)
                        (shaked ?s)
                        (not (unshaked ?s))
		        (not (contains ?s ?d1))
                        (not (contains ?s ?d2))))

  (:action pour-shot-to-used-shaker
           :parameters (?s - shot ?i - ingredient ?d - shaker ?h1 - hand ?l ?l1 - level)
           :precondition (and (contains ?s ?i)
                              (holding ?h1 ?s)
                              (unshaked ?d)
                              (shaker-level ?d ?l)
                              (next ?l ?l1))
           :effect (and (contains ?d ?i)
                        (shaker-level ?d ?l1)
	   	   	(empty ?s)
                        (not (contains ?s ?i))
  			(not (shaker-level ?d ?l))))

  (:action pour-shot-to-clean-shaker
           :parameters (?s - shot ?i - ingredient ?d - shaker ?h1 - hand ?l ?l1 - level)
           :precondition (and (contains ?s ?i)
                              (holding ?h1 ?s)
                              (empty ?d)
	   		      (clean ?d)
                              (shaker-level ?d ?l)
                              (next ?l ?l1))
           :effect (and (contains ?d ?i)
                        (unshaked ?d)
                        (shaker-level ?d ?l1)
	   	   	(empty ?s)
                        (not (contains ?s ?i))
                        (not (empty ?d))
			(not (clean ?d))
			(not (shaker-level ?d ?l))))

  (:action fill-shot
           :parameters (?s - shot ?i - ingredient ?h1 ?h2 - hand ?d - dispenser)
           :precondition (and (holding ?h1 ?s)
                              (empty ?s)
			      (clean ?s)
                              (handempty ?h2)
	   		      (dispenses ?d ?i))
           :effect (and (contains ?s ?i)
                        (used ?s ?i)
                        (not (empty ?s))
	   	   	(not (clean ?s))))

  (:action refill-shot
           :parameters (?s - shot ?i - ingredient ?h1 ?h2 - hand ?d - dispenser)
           :precondition (and (holding ?h1 ?s)
                              (empty ?s)
			      (used ?s ?i)
                              (handempty ?h2)
	   		      (dispenses ?d ?i))
           :effect (and (contains ?s ?i)
                        (not (empty ?s))))

  (:action grasp
             :parameters (?h - hand ?c - container)
             :precondition (and (ontable ?c) (handempty ?h))
             :effect (and (holding ?h ?c)
                          (not (ontable ?c))
	     	     	  (not (handempty ?h))))

  (:action leave
             :parameters (?h - hand ?c - container)
             :precondition (holding ?h ?c)
             :effect (and (ontable ?c)
                          (handempty ?h)
                          (not (holding ?h ?c))))

  (:action empty-shot
           :parameters (?h - hand ?p - shot ?b - beverage)
           :precondition (and (holding ?h ?p)
                              (contains ?p ?b))
           :effect (and (empty ?p)
                        (not (contains ?p ?b))))

  (:action clean-shot
  	   :parameters (?s - shot ?b - beverage ?h1 ?h2 - hand)
           :precondition (and (holding ?h1 ?s)
			      (empty ?s)
                              (used ?s ?b)
                              (handempty ?h2))
           :effect (and (clean ?s)
                        (not (used ?s ?b))))

  (:action empty-shaker
           :parameters (?h - hand ?s - shaker ?b - cocktail ?l ?l1 - level)
           :precondition (and (holding ?h ?s)
                              (contains ?s ?b)
			      (shaked ?s)
			      (shaker-level ?s ?l)
			      (shaker-empty-level ?s ?l1))
           :effect (and (empty ?s)
                        (shaker-level ?s ?l1)
                        (not (shaked ?s))
	   	   	(not (shaker-level ?s ?l))
			(not (contains ?s ?b))))

  (:action clean-shaker
  	   :parameters (?h1 ?h2 - hand ?s - shaker)
           :precondition (and (holding ?h1 ?s)
                              (empty ?s)
                              (handempty ?h2))
           :effect (and (clean ?s)))
