My reordering strategy places single-subsystem actions (grasp/leave operating on hand-container pairs) first, followed by shot-only actions, then shaker-only actions, and finally cross-subsystem actions (pour-shot-to-shaker, shake, pour-shaker-to-shot) last; predicates are reordered to group single-type predicates (container-only, shaker-only, hand-only) first and multi-type interaction predicates (contains, used, dispenses, cocktail-part, holding) last, with parameter ordering consistently placing the primary manipulated object first. This should help DecStar's causal graph analysis more quickly identify the star topology by making leaf-local variables and actions prominent early, potentially improving factoring quality and enabling more actions to be classified as leaf-only for the timeout instances.

(define (domain barman)
  (:requirements :strips :typing)
  (:types hand level beverage dispenser container - object
  	  ingredient cocktail - beverage
          shot shaker - container)
  (:predicates  (ontable ?c - container)
		(empty ?c - container)
		(clean ?c - container)
		(shaker-empty-level ?s - shaker ?l - level)
		(shaker-level ?s - shaker ?l - level)
		(unshaked ?s - shaker)
		(shaked ?s - shaker)
		(next ?l1 ?l2 - level)
		(handempty ?h - hand)
		(dispenses ?d - dispenser ?i - ingredient)
		(cocktail-part1 ?c - cocktail ?i - ingredient)
		(cocktail-part2 ?c - cocktail ?i - ingredient)
		(contains ?c - container ?b - beverage)
		(used ?c - container ?b - beverage)
		(holding ?h - hand ?c - container))

  (:action grasp
             :parameters (?h - hand ?c - container)
             :precondition (and (ontable ?c) (handempty ?h))
             :effect (and (not (ontable ?c))
	     	     	  (not (handempty ?h))
			  (holding ?h ?c)))

  (:action leave
             :parameters (?h - hand ?c - container)
             :precondition (holding ?h ?c)
             :effect (and (not (holding ?h ?c))
	     	     	  (handempty ?h)
			  (ontable ?c)))

  (:action empty-shot
           :parameters (?p - shot ?b - beverage ?h - hand)
           :precondition (and (contains ?p ?b)
                              (holding ?h ?p))
           :effect (and (not (contains ?p ?b))
	   	   	(empty ?p)))

  (:action clean-shot
  	   :parameters (?s - shot ?b - beverage ?h1 ?h2 - hand)
           :precondition (and (empty ?s)
                              (used ?s ?b)
                              (holding ?h1 ?s)
                              (handempty ?h2))
           :effect (and (not (used ?s ?b))
	   	   	(clean ?s)))

  (:action fill-shot
           :parameters (?s - shot ?i - ingredient ?d - dispenser ?h1 ?h2 - hand)
           :precondition (and (empty ?s)
			      (clean ?s)
                              (holding ?h1 ?s)
                              (dispenses ?d ?i)
                              (handempty ?h2))
           :effect (and (not (empty ?s))
	   	   	(not (clean ?s))
			(contains ?s ?i)
			(used ?s ?i)))

  (:action refill-shot
           :parameters (?s - shot ?i - ingredient ?d - dispenser ?h1 ?h2 - hand)
           :precondition (and (empty ?s)
			      (used ?s ?i)
                              (holding ?h1 ?s)
                              (dispenses ?d ?i)
                              (handempty ?h2))
           :effect (and (not (empty ?s))
                        (contains ?s ?i)))

  (:action clean-shaker
  	   :parameters (?s - shaker ?h1 ?h2 - hand)
           :precondition (and (empty ?s)
                              (holding ?h1 ?s)
                              (handempty ?h2))
           :effect (and (clean ?s)))

  (:action empty-shaker
           :parameters (?s - shaker ?b - cocktail ?l ?l1 - level ?h - hand)
           :precondition (and (contains ?s ?b)
			      (shaked ?s)
			      (shaker-level ?s ?l)
			      (shaker-empty-level ?s ?l1)
                              (holding ?h ?s))
           :effect (and (not (shaked ?s))
	   	   	(not (shaker-level ?s ?l))
	   	   	(shaker-level ?s ?l1)
			(not (contains ?s ?b))
	   	   	(empty ?s)))

  (:action pour-shot-to-clean-shaker
           :parameters (?s - shot ?i - ingredient ?d - shaker ?h1 - hand ?l ?l1 - level)
           :precondition (and (contains ?s ?i)
                              (holding ?h1 ?s)
                              (empty ?d)
	   		      (clean ?d)
                              (shaker-level ?d ?l)
                              (next ?l ?l1))
           :effect (and (not (contains ?s ?i))
	   	   	(empty ?s)
			(contains ?d ?i)
                        (not (empty ?d))
			(not (clean ?d))
			(unshaked ?d)
			(not (shaker-level ?d ?l))
			(shaker-level ?d ?l1)))

  (:action pour-shot-to-used-shaker
           :parameters (?s - shot ?i - ingredient ?d - shaker ?h1 - hand ?l ?l1 - level)
           :precondition (and (contains ?s ?i)
                              (holding ?h1 ?s)
                              (unshaked ?d)
                              (shaker-level ?d ?l)
                              (next ?l ?l1))
           :effect (and (not (contains ?s ?i))
	   	   	(empty ?s)
                        (contains ?d ?i)
  			(not (shaker-level ?d ?l))
			(shaker-level ?d ?l1)))

  (:action shake
  	   :parameters (?s - shaker ?b - cocktail ?d1 ?d2 - ingredient ?h1 ?h2 - hand)
           :precondition (and (contains ?s ?d1)
                              (contains ?s ?d2)
                              (unshaked ?s)
                              (holding ?h1 ?s)
                              (cocktail-part1 ?b ?d1)
			      (cocktail-part2 ?b ?d2)
                              (handempty ?h2))
           :effect (and (not (unshaked ?s))
		        (not (contains ?s ?d1))
                        (not (contains ?s ?d2))
	   	   	(shaked ?s)
                        (contains ?s ?b)))

  (:action pour-shaker-to-shot
           :parameters (?s - shaker ?b - beverage ?d - shot ?h - hand ?l ?l1 - level)
           :precondition (and (contains ?s ?b)
			      (shaked ?s)
                              (shaker-level ?s ?l)
                              (holding ?h ?s)
                              (next ?l1 ?l)
                              (empty ?d)
			      (clean ?d))
           :effect (and (not (shaker-level ?s ?l))
			(shaker-level ?s ?l1)
			(not (clean ?d))
	   	   	(not (empty ?d))
			(contains ?d ?b)))
