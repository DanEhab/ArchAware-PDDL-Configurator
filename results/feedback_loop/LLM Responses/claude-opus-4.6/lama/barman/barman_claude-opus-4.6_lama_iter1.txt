My reordering strategy focuses on further optimizing precondition ordering within actions to place static predicates first and most restrictive conditions early, reordering effects to prioritize goal-relevant add-effects before auxiliary ones and delete-effects last, and fine-tuning action ordering to place simpler enabling actions earlier while keeping the predicate block optimized with goal-relevant predicates first and static predicates last. I expect this to improve FF's relaxed-plan extraction quality (better preferred operators) and accelerate applicability checking, particularly helping on the harder instances (04, 13) where IPC gains were negative.

(define (domain barman)
  (:requirements :strips :typing)
  (:types hand level beverage dispenser container - object
  	  ingredient cocktail - beverage
          shot shaker - container)
  (:predicates  (contains ?c - container ?b - beverage)
                (holding ?h - hand ?c - container)
                (empty ?c - container)
                (clean ?c - container)
                (handempty ?h - hand)
                (ontable ?c - container)
                (used ?c - container ?b - beverage)
                (shaked ?s - shaker)
                (unshaked ?s - shaker)
                (shaker-level ?s - shaker ?l - level)
                (shaker-empty-level ?s - shaker ?l - level)
                (next ?l1 ?l2 - level)
                (dispenses ?d - dispenser ?i - ingredient)
                (cocktail-part1 ?c - cocktail ?i - ingredient)
                (cocktail-part2 ?c - cocktail ?i - ingredient))

  (:action grasp
             :parameters (?h - hand ?c - container)
             :precondition (and (ontable ?c) (handempty ?h))
             :effect (and (holding ?h ?c)
                          (not (ontable ?c))
	     	     	  (not (handempty ?h))))

  (:action leave
             :parameters (?h - hand ?c - container)
             :precondition (holding ?h ?c)
             :effect (and (ontable ?c)
                          (handempty ?h)
                          (not (holding ?h ?c))))

  (:action fill-shot
           :parameters (?s - shot ?i - ingredient ?h1 ?h2 - hand ?d - dispenser)
           :precondition (and (dispenses ?d ?i)
                              (holding ?h1 ?s)
                              (empty ?s)
                              (clean ?s)
                              (handempty ?h2))
           :effect (and (contains ?s ?i)
                        (used ?s ?i)
                        (not (clean ?s))
	   	   	(not (empty ?s))))

  (:action refill-shot
           :parameters (?s - shot ?i - ingredient ?h1 ?h2 - hand ?d - dispenser)
           :precondition (and (dispenses ?d ?i)
                              (holding ?h1 ?s)
                              (used ?s ?i)
                              (empty ?s)
                              (handempty ?h2))
           :effect (and (contains ?s ?i)
                        (not (empty ?s))))

  (:action empty-shot
           :parameters (?h - hand ?p - shot ?b - beverage)
           :precondition (and (holding ?h ?p)
                              (contains ?p ?b))
           :effect (and (empty ?p)
                        (not (contains ?p ?b))))

  (:action clean-shot
  	   :parameters (?s - shot ?b - beverage ?h1 ?h2 - hand)
           :precondition (and (holding ?h1 ?s)
                              (used ?s ?b)
			      (empty ?s)
                              (handempty ?h2))
           :effect (and (clean ?s)
                        (not (used ?s ?b))))

  (:action clean-shaker
  	   :parameters (?h1 ?h2 - hand ?s - shaker)
           :precondition (and (holding ?h1 ?s)
                              (empty ?s)
                              (handempty ?h2))
           :effect (and (clean ?s)))

  (:action pour-shot-to-clean-shaker
           :parameters (?s - shot ?i - ingredient ?d - shaker ?h1 - hand ?l ?l1 - level)
           :precondition (and (next ?l ?l1)
                              (holding ?h1 ?s)
			      (contains ?s ?i)
                              (empty ?d)
                              (clean ?d)
                              (shaker-level ?d ?l))
           :effect (and (contains ?d ?i)
                        (unshaked ?d)
                        (shaker-level ?d ?l1)
                        (empty ?s)
                        (not (contains ?s ?i))
                        (not (empty ?d))
			(not (clean ?d))
			(not (shaker-level ?d ?l))))

  (:action pour-shot-to-used-shaker
           :parameters (?s - shot ?i - ingredient ?d - shaker ?h1 - hand ?l ?l1 - level)
           :precondition (and (next ?l ?l1)
                              (holding ?h1 ?s)
			      (contains ?s ?i)
                              (unshaked ?d)
                              (shaker-level ?d ?l))
           :effect (and (contains ?d ?i)
                        (shaker-level ?d ?l1)
                        (empty ?s)
                        (not (contains ?s ?i))
  			(not (shaker-level ?d ?l))))

  (:action shake
  	   :parameters (?b - cocktail ?d1 ?d2 - ingredient ?s - shaker ?h1 ?h2 - hand)
           :precondition (and (cocktail-part1 ?b ?d1)
			      (cocktail-part2 ?b ?d2)
                              (holding ?h1 ?s)
			      (contains ?s ?d1)
                              (contains ?s ?d2)
			      (unshaked ?s)
                              (handempty ?h2))
           :effect (and (shaked ?s)
                        (contains ?s ?b)
                        (not (unshaked ?s))
		        (not (contains ?s ?d1))
                        (not (contains ?s ?d2))))

  (:action pour-shaker-to-shot
           :parameters (?b - beverage ?d - shot ?h - hand ?s - shaker ?l ?l1 - level)
           :precondition (and (next ?l1 ?l)
                              (holding ?h ?s)
			      (shaked ?s)
			      (contains ?s ?b)
			      (empty ?d)
			      (clean ?d)
                              (shaker-level ?s ?l))
           :effect (and (contains ?d ?b)
                        (shaker-level ?s ?l1)
                        (not (shaker-level ?s ?l))
                        (not (clean ?d))
	   	   	(not (empty ?d))))

  (:action empty-shaker
           :parameters (?h - hand ?s - shaker ?b - cocktail ?l ?l1 - level)
           :precondition (and (shaker-empty-level ?s ?l1)
                              (holding ?h ?s)
			      (shaked ?s)
                              (contains ?s ?b)
			      (shaker-level ?s ?l))
           :effect (and (empty ?s)
                        (shaker-level ?s ?l1)
                        (not (shaked ?s))
	   	   	(not (shaker-level ?s ?l))
			(not (contains ?s ?b))))
