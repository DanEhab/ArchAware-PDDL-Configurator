To optimize for LAMA's architecture, I ordered predicates by placing the goal-relevant `at` and frequently updated `free` first to accelerate mutex discovery, while placing static predicates `NEXT` and `BLOCKED` last. I also ordered actions by precondition count (simplest first) and placed static preconditions early in each action to optimize Fast Downward's successor generation and relaxed-plan extraction.

(define (domain ricochet-robots)
(:requirements :strips :typing :negative-preconditions)

(:types
    robot - object
    cell - object
    direction - object
)

(:predicates
    (at ?r - robot ?c - cell)
    (free ?c - cell)
    (is-moving ?r - robot ?dir - direction)
    (nothing-is-moving)
    (NEXT ?c - cell ?cnext - cell ?dir - direction)
    (BLOCKED ?c - cell ?dir - direction)
)

(:action go
    :parameters (?r - robot ?dir - direction)
    :precondition 
        (and
            (nothing-is-moving)
        )
    :effect 
        (and
            (is-moving ?r ?dir)
            (not (nothing-is-moving))
        )
)

(:action stop-at-barrier
    :parameters (?r - robot ?cat - cell ?dir - direction)
    :precondition 
        (and
            (BLOCKED ?cat ?dir)
            (at ?r ?cat)
            (is-moving ?r ?dir)
        )
    :effect 
        (and
            (nothing-is-moving)
            (not (is-moving ?r ?dir))
        )
)

(:action stop-at-robot
    :parameters (?r - robot ?cat - cell ?cnext - cell ?dir - direction)
    :precondition 
        (and
            (NEXT ?cat ?cnext ?dir)
            (at ?r ?cat)
            (is-moving ?r ?dir)
            (not (free ?cnext))
        )
    :effect 
        (and
            (nothing-is-moving)
            (not (is-moving ?r ?dir))
        )
)

(:action step
    :parameters (?r - robot ?cfrom - cell ?cto - cell ?dir - direction)
    :precondition 
        (and
            (NEXT ?cfrom ?cto ?dir)
            (not (BLOCKED ?cfrom ?dir))
            (at ?r ?cfrom)
            (is-moving ?r ?dir)
            (free ?cto)
        )
    :effect 
        (and
            (at ?r ?cto)
            (free ?cfrom)
            (not (at ?r ?cfrom))
            (not (free ?cto))
        )
)
)
