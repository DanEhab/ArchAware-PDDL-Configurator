(define (domain barman)
  (:requirements :strips :typing)
  (:types hand level beverage dispenser container - object
  	  ingredient cocktail - beverage
          shot shaker - container)
  (:predicates  (ontable ?c - container)
                (holding ?h - hand ?c - container)
		(handempty ?h - hand)
		(empty ?c - container)
                (contains ?c - container ?b - beverage)
		(clean ?c - container)
                (used ?c - container ?b - beverage)
                (dispenses ?d - dispenser ?i - ingredient)
		(shaker-empty-level ?s - shaker ?l - level)
		(shaker-level ?s - shaker ?l - level)
		(next ?l1 ?l2 - level)
		(unshaked ?s - shaker)
		(shaked ?s - shaker)
                (cocktail-part1 ?c - cocktail ?i - ingredient)
                (cocktail-part2 ?c - cocktail ?i - ingredient))
		
  (:action grasp
             :parameters (?h - hand ?c - container)
             :precondition (and (ontable ?c) (handempty ?h))
             :effect (and (not (ontable ?c))
	     	     	  (not (handempty ?h))
			  (holding ?h ?c)))

  (:action leave
             :parameters (?h - hand ?c - container)
             :precondition (holding ?h ?c)
             :effect (and (not (holding ?h ?c))
	     	     	  (handempty ?h)
			  (ontable ?c)))
  
  (:action fill-shot
           :parameters (?s - shot ?i - ingredient ?h1 ?h2 - hand ?d - dispenser)
           :precondition (and (clean ?s)
                              (dispenses ?d ?i)
                              (empty ?s)
                              (handempty ?h2)
                              (holding ?h1 ?s))
           :effect (and (contains ?s ?i)
                        (not (clean ?s))
                        (not (empty ?s))
                        (used ?s ?i)))

  (:action refill-shot
           :parameters (?s - shot ?i - ingredient ?h1 ?h2 - hand ?d - dispenser)
           :precondition (and (dispenses ?d ?i)
                              (empty ?s)
                              (handempty ?h2)
                              (holding ?h1 ?s)
                              (used ?s ?i))
           :effect (and (contains ?s ?i)
                        (not (empty ?s))))

  (:action empty-shot
           :parameters (?h - hand ?p - shot ?b - beverage)
           :precondition (and (contains ?p ?b)
                              (holding ?h ?p))
           :effect (and (empty ?p)
                        (not (contains ?p ?b))))

  (:action clean-shot
  	   :parameters (?s - shot ?b - beverage ?h1 ?h2 - hand)
           :precondition (and (empty ?s)
                              (handempty ?h2)
                              (holding ?h1 ?s)
                              (used ?s ?b))
           :effect (and (clean ?s)
                        (not (used ?s ?b))))

  (:action pour-shot-to-clean-shaker
           :parameters (?s - shot ?i - ingredient ?d - shaker ?h1 - hand ?l ?l1 - level)
           :precondition (and (clean ?d)
                              (contains ?s ?i)
                              (empty ?d)
                              (holding ?h1 ?s)
                              (next ?l ?l1)
                              (shaker-level ?d ?l))
           :effect (and (contains ?d ?i)
                        (empty ?s)
                        (not (clean ?d))
                        (not (contains ?s ?i))
                        (not (empty ?d))
                        (not (shaker-level ?d ?l))
                        (shaker-level ?d ?l1)
                        (unshaked ?d)))

  (:action pour-shot-to-used-shaker
           :parameters (?s - shot ?i - ingredient ?d - shaker ?h1 - hand ?l ?l1 - level)
           :precondition (and (contains ?s ?i)
                              (holding ?h1 ?s)
                              (next ?l ?l1)
                              (shaker-level ?d ?l)
                              (unshaked ?d))
           :effect (and (contains ?d ?i)
                        (empty ?s)
                        (not (contains ?s ?i))
                        (not (shaker-level ?d ?l))
                        (shaker-level ?d ?l1)))

  (:action empty-shaker
           :parameters (?h - hand ?s - shaker ?b - cocktail ?l ?l1 - level)
           :precondition (and (contains ?s ?b)
                              (holding ?h ?s)
                              (shaker-empty-level ?s ?l1)
                              (shaker-level ?s ?l)
                              (shaked ?s))
           :effect (and (empty ?s)
                        (not (contains ?s ?b))
                        (not (shaked ?s))
                        (not (shaker-level ?s ?l))
                        (shaker-level ?s ?l1)))

  (:action clean-shaker
  	   :parameters (?h1 ?h2 - hand ?s - shaker)
           :precondition (and (empty ?s)
                              (handempty ?h2)
                              (holding ?h1 ?s))
           :effect (and (clean ?s)))
  
  (:action shake
  	   :parameters (?b - cocktail ?d1 ?d2 - ingredient ?s - shaker ?h1 ?h2 - hand)
           :precondition (and (cocktail-part1 ?b ?d1)
                              (cocktail-part2 ?b ?d2)
                              (contains ?s ?d1)
                              (contains ?s ?d2)
                              (handempty ?h2)
                              (holding ?h1 ?s)
                              (unshaked ?s))
           :effect (and (contains ?s ?b)
                        (not (contains ?s ?d1))
                        (not (contains ?s ?d2))
                        (not (unshaked ?s))
                        (shaked ?s)))

  (:action pour-shaker-to-shot
           :parameters (?b - beverage ?d - shot ?h - hand ?s - shaker ?l ?l1 - level)
           :precondition (and (clean ?d)
                              (contains ?s ?b)
                              (empty ?d)
                              (holding ?h ?s)
                              (next ?l1 ?l)
                              (shaker-level ?s ?l)
                              (shaked ?s))
           :effect (and (contains ?d ?b)
                        (not (clean ?d))
                        (not (empty ?d))
                        (not (shaker-level ?s ?l))
                        (shaker-level ?s ?l1)))
 )
